def solve_n_queens(n):
    board = [-1] * n   # board[i] = column position of queen in row i

    # To keep track of attacks (Branch & Bound)
    cols = [False] * n
    diag1 = [False] * (2*n)   # row - col + n
    diag2 = [False] * (2*n)   # row + col

    def backtrack(row):
        if row == n:
            return True
        
        for col in range(n):
            #Check constraints (Branch & Bound pruning)
            if cols[col] or diag1[row - col + n] or diag2[row + col]:
                continue
            
            #Place queen
            board[row] = col
            cols[col] = diag1[row - col + n] = diag2[row + col] = True
            
            #Recursionn
            if backtrack(row + 1):
                return True
            
            #Backtrack
            cols[col] = diag1[row - col + n] = diag2[row + col] = False
        
        return False

    if backtrack(0):
        return board
    else:
        return None


# Input from user
n = int(input("Enter value of n: "))

solution = solve_n_queens(n)

# Print solution
if solution:
    print("\nSolution (row → column):", solution)
    
    print("\nBoard:")
    for i in range(n):
        row = ['.'] * n
        row[solution[i]] = 'Q'
        print(' '.join(row))
else:
    print("No solution exists")
